🔗 C INTEGRATION EXAMPLE

#include <stdio.h>
#include "jules_ffi.h"

int main() {
    // Initialize Jules
    JulesContext *ctx = julius_init();
    
    // Create tensors for ML
    usize shape[] = {4, 1000, 1000};  // 4x1000x1000 matrix
    JulesTensor *t = julius_tensor_create(shape, 3);
    
    // Create physics bodies
    for (int i = 0; i < 10; i++) {
        uint64_t body = julius_physics_body_create(
            ctx, 1.0,           // mass
            (float)i * 2, 10.0, // x, y
            0.0                 // z
        );
    }
    
    // Step simulation 1000 times
    for (int step = 0; step < 1000; step++) {
        julius_physics_step(ctx, 0.016);  // 16ms timestep
    }
    
    // Cleanup
    julius_tensor_destroy(t);
    julius_destroy(ctx);
    
    return 0;
}
